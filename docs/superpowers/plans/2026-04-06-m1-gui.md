```rust
use types::{NodeId, Vec2};

#[derive(Debug, Clone)]
pub struct Viewport {
    pub offset: Vec2,
    pub zoom: f32,
}

impl Default for Viewport {
    fn default() -> Self {
        Self {
            offset: Vec2::new(0.0, 0.0),
            zoom: 1.0,
        }
    }
}

impl Viewport {
    /// 画布坐标 → 屏幕坐标
    pub fn to_screen(&self, canvas_pos: Vec2) -> Vec2 {
        Vec2::new(
            (canvas_pos.x + self.offset.x) * self.zoom,
            (canvas_pos.y + self.offset.y) * self.zoom,
        )
    }

    /// 屏幕坐标 → 画布坐标
    pub fn to_canvas(&self, screen_pos: Vec2) -> Vec2 {
        Vec2::new(
            screen_pos.x / self.zoom - self.offset.x,
            screen_pos.y / self.zoom - self.offset.y,
        )
    }

    /// 以某点为中心缩放
    pub fn zoom_at(&mut self, center: Vec2, factor: f32) {
        let canvas_center = self.to_canvas(center);
        self.zoom = (self.zoom * factor).clamp(0.1, 5.0);
        self.offset.x = center.x / self.zoom - canvas_center.x;
        self.offset.y = center.y / self.zoom - canvas_center.y;
    }
}

#[derive(Debug, Clone)]
pub enum ActiveInteraction {
    Dragging {
        node: NodeId,
        start_pos: Vec2,
    },
    Connecting {
        from_node: NodeId,
        from_pin: String,
        cursor: Vec2,
    },
    Panning {
        start_offset: Vec2,
        start_cursor: Vec2,
    },
}

#[derive(Debug, Clone, PartialEq)]
pub enum HoverTarget {
    Node(NodeId),
    InputPin(NodeId, String),
    OutputPin(NodeId, String),
}

#[derive(Debug, Clone)]
pub struct CanvasState {
    pub viewport: Viewport,
    pub interaction: Option<ActiveInteraction>,
    pub hover: Option<HoverTarget>,
}

impl Default for CanvasState {
    fn default() -> Self {
        Self {
            viewport: Viewport::default(),
            interaction: None,
            hover: None,
        }
    }
}
```

```rust
use iced::widget::canvas::Frame;
use iced::{Point, Size, Color};

use crate::canvas::Viewport;
use super::GRID_DOT;

pub fn draw_grid(frame: &mut Frame, viewport: &Viewport, bounds: Size) {
    let spacing = 24.0 * viewport.zoom;
    if spacing < 4.0 {
        return; // 缩放太小不画网格
    }

    let offset_x = (viewport.offset.x * viewport.zoom) % spacing;
    let offset_y = (viewport.offset.y * viewport.zoom) % spacing;

    let dot_radius = 1.0;
    let mut x = offset_x;
    while x < bounds.width {
        let mut y = offset_y;
        while y < bounds.height {
            let dot = iced::widget::canvas::Path::circle(Point::new(x, y), dot_radius);
            frame.fill(&dot, GRID_DOT);
            y += spacing;
        }
        x += spacing;
    }
}
```

```rust
use iced::widget::canvas::Frame;
use iced::{Point, Size, Color, Rectangle};

use engine::registry::def::{NodeDef, ParamDef};
use types::{NodeId, Value, Vec2};

use crate::canvas::Viewport;
use super::{NODE_BG, BORDER, BORDER_SELECTED, TEXT_PRIMARY, TEXT_SECONDARY, TEXT_MUTED, pin_color, category_color};

pub const NODE_WIDTH: f32 = 192.0;
pub const NODE_RADIUS: f32 = 8.0;
pub const HEADER_HEIGHT: f32 = 32.0;
pub const PIN_RADIUS: f32 = 5.0;
pub const PADDING: f32 = 10.0;
pub const PARAM_HEIGHT: f32 = 28.0;
pub const PIN_ROW_HEIGHT: f32 = 20.0;

/// 计算节点的屏幕矩形
pub fn node_bounds(
    position: Vec2,
    def: &NodeDef,
    viewport: &Viewport,
) -> Rectangle {
    let screen = viewport.to_screen(position);
    let w = NODE_WIDTH * viewport.zoom;
    let pin_rows = def.inputs.len().max(def.outputs.len());
    let param_rows = def.params.len();
    let h = (HEADER_HEIGHT + (pin_rows as f32) * PIN_ROW_HEIGHT + (param_rows as f32) * PARAM_HEIGHT + PADDING * 2.0) * viewport.zoom;
    Rectangle::new(Point::new(screen.x, screen.y), Size::new(w, h))
}

/// 计算输入引脚的屏幕坐标
pub fn input_pin_position(
    node_pos: Vec2,
    pin_index: usize,
    viewport: &Viewport,
) -> Point {
    let screen = viewport.to_screen(node_pos);
    let y = screen.y + (HEADER_HEIGHT + PADDING + pin_index as f32 * PIN_ROW_HEIGHT + PIN_ROW_HEIGHT / 2.0) * viewport.zoom;
    Point::new(screen.x, y)
}

/// 计算输出引脚的屏幕坐标
pub fn output_pin_position(
    node_pos: Vec2,
    pin_index: usize,
    num_inputs: usize,
    viewport: &Viewport,
) -> Point {
    let screen = viewport.to_screen(node_pos);
    let w = NODE_WIDTH * viewport.zoom;
    let y = screen.y + (HEADER_HEIGHT + PADDING + pin_index as f32 * PIN_ROW_HEIGHT + PIN_ROW_HEIGHT / 2.0) * viewport.zoom;
    Point::new(screen.x + w, y)
}

/// 绘制单个节点
pub fn draw_node(
    frame: &mut Frame,
    position: Vec2,
    def: &NodeDef,
    params: &std::collections::HashMap<String, Value>,
    selected: bool,
    viewport: &Viewport,
) {
    let bounds = node_bounds(position, def, viewport);
    let z = viewport.zoom;

    // 节点背景
    let bg = iced::widget::canvas::Path::rounded_rectangle(
        bounds.position(),
        bounds.size(),
        (NODE_RADIUS * z).into(),
    );
    frame.fill(&bg, NODE_BG);

    // 边框
    let border_color = if selected { BORDER_SELECTED } else { BORDER };
    let stroke = iced::widget::canvas::Stroke::default()
        .with_color(border_color)
        .with_width(if selected { 1.5 } else { 1.0 });
    frame.stroke(&bg, stroke);

    // Header 背景
    let header_bg = iced::widget::canvas::Path::new(|builder| {
        let p = bounds.position();
        let r = NODE_RADIUS * z;
        let w = bounds.width;
        let h = HEADER_HEIGHT * z;
        builder.move_to(Point::new(p.x + r, p.y));
        builder.line_to(Point::new(p.x + w - r, p.y));
        builder.arc_to(Point::new(p.x + w, p.y), Point::new(p.x + w, p.y + r), r);
        builder.line_to(Point::new(p.x + w, p.y + h));
        builder.line_to(Point::new(p.x, p.y + h));
        builder.line_to(Point::new(p.x, p.y + r));
        builder.arc_to(Point::new(p.x, p.y), Point::new(p.x + r, p.y), r);
        builder.close();
    });
    frame.fill(&header_bg, category_color(&def.category));

    // Header 文字
    frame.fill_text(iced::widget::canvas::Text {
        content: def.name.clone(),
        position: Point::new(bounds.x + PADDING * z, bounds.y + 8.0 * z),
        size: (13.0 * z).into(),
        color: Color::WHITE,
        ..Default::default()
    });

    // 输入引脚
    for (i, pin) in def.inputs.iter().enumerate() {
        let pos = input_pin_position(position, i, viewport);
        let circle = iced::widget::canvas::Path::circle(pos, PIN_RADIUS * z);
        frame.fill(&circle, pin_color(&pin.data_type.0));

        frame.fill_text(iced::widget::canvas::Text {
            content: pin.name.clone(),
            position: Point::new(pos.x + (PIN_RADIUS + 6.0) * z, pos.y - 5.0 * z),
            size: (11.0 * z).into(),
            color: TEXT_SECONDARY,
            ..Default::default()
        });
    }

    // 输出引脚
    for (i, pin) in def.outputs.iter().enumerate() {
        let pos = output_pin_position(position, i, def.inputs.len(), viewport);
        let circle = iced::widget::canvas::Path::circle(pos, PIN_RADIUS * z);
        frame.fill(&circle, pin_color(&pin.data_type.0));

        frame.fill_text(iced::widget::canvas::Text {
            content: pin.name.clone(),
            position: Point::new(pos.x - (PIN_RADIUS + 6.0) * z - 40.0 * z, pos.y - 5.0 * z),
            size: (11.0 * z).into(),
            color: TEXT_SECONDARY,
            horizontal_alignment: iced::alignment::Horizontal::Right,
            ..Default::default()
        });
    }
}
```

```rust
use iced::widget::canvas::{Frame, Path, Stroke};
use iced::Point;

use super::pin_color;

/// 绘制贝塞尔连线
pub fn draw_connection(
    frame: &mut Frame,
    from: Point,
    to: Point,
    data_type: &str,
    _selected: bool,
) {
    let dx = (to.x - from.x).abs() * 0.4;
    let path = Path::new(|builder| {
        builder.move_to(from);
        builder.bezier_curve_to(
            Point::new(from.x + dx, from.y),
            Point::new(to.x - dx, to.y),
            to,
        );
    });

    let stroke = Stroke::default()
        .with_color(pin_color(data_type))
        .with_width(1.5);
    frame.stroke(&path, stroke);
}

/// 绘制临时连线（拖拽中）
pub fn draw_temp_connection(
    frame: &mut Frame,

    to: Point,

        builder.move_to(from);
        builder.bezier_curve_to(
            Point::new(from.x + dx, from.y),
            Point::new(to.x - dx, to.y),
            to,
        );
    });

    let stroke = Stroke::default()
        .with_color(super::TEXT_MUTED)
```

```rust
pub mod state;
pub mod controller;

pub use state::{CanvasState, Viewport, ActiveInteraction, HoverTarget};

use std::collections::{HashMap, HashSet};
use std::sync::Arc;

use iced::mouse;
use iced::widget::canvas;
use iced::widget::canvas::event;
use iced::{Rectangle, Theme};

use engine::graph::Graph;
use engine::registry::NodeManager;
use types::{NodeId, Value};

use crate::render;

/// NodeCanvas 的外部引用数据（每帧从 App 传入）
pub struct NodeCanvasData<'a> {
    pub graph: &'a Graph,
    pub node_manager: &'a Arc<NodeManager>,
    pub selection: &'a HashSet<NodeId>,
    pub canvas_state: &'a CanvasState,
}

pub struct NodeCanvas<'a> {
    data: NodeCanvasData<'a>,
}

impl<'a> NodeCanvas<'a> {
    pub fn new(data: NodeCanvasData<'a>) -> Self {
        Self { data }
    }
}

#[derive(Default)]
pub struct CanvasCache {
    cache: canvas::Cache,
}

impl<'a> canvas::Program<crate::Message> for NodeCanvas<'a> {
    type State = CanvasCache;

    fn update(
        &self,
        _state: &mut Self::State,
        event: canvas::Event,
        bounds: Rectangle,
        cursor: mouse::Cursor,
    ) -> (event::Status, Option<crate::Message>) {
        if let Some(pos) = cursor.position_in(bounds) {
            return (
                event::Status::Captured,
                Some(crate::Message::CanvasEvent(event, pos.x, pos.y)),
            );
        }
        (event::Status::Ignored, None)
    }

    fn draw(
        &self,
        state: &Self::State,
        renderer: &iced::Renderer,
        _theme: &Theme,
        bounds: Rectangle,
        cursor: mouse::Cursor,
    ) -> Vec<canvas::Geometry> {
        let geom = state.cache.draw(renderer, bounds.size(), |frame| {
            let vp = &self.data.canvas_state.viewport;

            // 背景网格
            render::background::draw_grid(frame, vp, bounds.size());

            // 连线
            for conn in &self.data.graph.connections {
                let from_node = self.data.graph.nodes.get(&conn.from_node);
                let to_node = self.data.graph.nodes.get(&conn.to_node);
                if let (Some(from_n), Some(to_n)) = (from_node, to_node) {
                    let from_def = self.data.node_manager.get(&from_n.type_id);
                    let to_def = self.data.node_manager.get(&to_n.type_id);
                    if let (Some(fd), Some(td)) = (from_def, to_def) {
                        let from_idx = fd.outputs.iter().position(|p| p.name == conn.from_pin).unwrap_or(0);
                        let to_idx = td.inputs.iter().position(|p| p.name == conn.to_pin).unwrap_or(0);
                        let from_pt = render::node::output_pin_position(from_n.position, from_idx, fd.inputs.len(), vp);
                        let to_pt = render::node::input_pin_position(to_n.position, to_idx, vp);
                        let dt = fd.outputs.get(from_idx).map(|p| p.data_type.0.as_str()).unwrap_or("string");
                        render::connection::draw_connection(frame, from_pt, to_pt, dt, false);
                    }
                }
            }

            // 节点
            for (id, node) in &self.data.graph.nodes {
                if let Some(def) = self.data.node_manager.get(&node.type_id) {
                    let selected = self.data.selection.contains(id);
                    render::node::draw_node(frame, node.position, def, &node.params, selected, vp);
                }
            }

            // 临时连线
            if let Some(ActiveInteraction::Connecting { from_node, from_pin, cursor: cur }) = &self.data.canvas_state.interaction {
                if let Some(node) = self.data.graph.nodes.get(from_node) {
                    if let Some(def) = self.data.node_manager.get(&node.type_id) {
                        let idx = def.outputs.iter().position(|p| p.name == *from_pin).unwrap_or(0);
                        let from_pt = render::node::output_pin_position(node.position, idx, def.inputs.len(), vp);
                        let to_pt = iced::Point::new(cur.x, cur.y);
                        render::connection::draw_temp_connection(frame, from_pt, to_pt);
                    }
                }
            }
        });

        vec![geom]
    }

    fn mouse_interaction(
        &self,
        _state: &Self::State,
        _bounds: Rectangle,
        _cursor: mouse::Cursor,
    ) -> mouse::Interaction {
        match &self.data.canvas_state.interaction {
            Some(ActiveInteraction::Panning { .. }) => mouse::Interaction::Grabbing,
            Some(ActiveInteraction::Dragging { .. }) => mouse::Interaction::Grabbing,
            _ => mouse::Interaction::default(),
        }
    }
}
```

```rust
use crate::render::node::{node_bounds, input_pin_position, output_pin_position, PIN_RADIUS, NODE_WIDTH};
use crate::Message;

pub struct CanvasController;

impl CanvasController {
    /// 处理画布事件，返回需要发出的消息
    pub fn handle_event(
        event: &canvas::Event,
        cursor_x: f32,
        cursor_y: f32,
        canvas_state: &mut CanvasState,
        graph: &Graph,
        node_manager: &Arc<NodeManager>,
        selection: &mut HashSet<NodeId>,
        graph_controller: &mut GraphController,
    ) -> Vec<Message> {
        let mut messages = Vec::new();

        match event {
            canvas::Event::Mouse(mouse_event) => {
                Self::handle_mouse(
                    mouse_event,
                    cursor_x, cursor_y,
                    canvas_state,
                    graph, node_manager, selection, graph_controller,
                    &mut messages,
                );
            }
            _ => {}
        }

        // 状态变化后清缓存
        messages
    }

    fn handle_mouse(
        event: &iced::mouse::Event,
        cx: f32, cy: f32,
        state: &mut CanvasState,
        graph: &Graph,
        nm: &Arc<NodeManager>,
        selection: &mut HashSet<NodeId>,
        gc: &mut GraphController,
        msgs: &mut Vec<Message>,
    ) {
        use iced::mouse::Event::*;
        use iced::mouse::Button;

        match event {
            ButtonPressed(Button::Left) => {
                let canvas_pos = state.viewport.to_canvas(Vec2::new(cx, cy));

                // 检查是否点击了输出引脚 → 开始连线
                if let Some((node_id, pin_name)) = Self::hit_output_pin(cx, cy, graph, nm, &state.viewport) {
                    state.interaction = Some(ActiveInteraction::Connecting {
                        from_node: node_id,
                        from_pin: pin_name,
                        cursor: Vec2::new(cx, cy),
                    });
                    return;
                }

                // 检查是否点击了节点 → 选中 + 开始拖拽
                if let Some(node_id) = Self::hit_node(cx, cy, graph, nm, &state.viewport) {
                    if !selection.contains(&node_id) {
                        selection.clear();
                        selection.insert(node_id);
                    }
                    state.interaction = Some(ActiveInteraction::Dragging {
                        node: node_id,
                        start_pos: canvas_pos,
                    });
                    return;
                }

                // 点击空白 → 清选中 + 开始平移
                selection.clear();
                state.interaction = Some(ActiveInteraction::Panning {
                    start_offset: state.viewport.offset,
                    start_cursor: Vec2::new(cx, cy),
                });
            }

            ButtonPressed(Button::Middle) => {
                state.interaction = Some(ActiveInteraction::Panning {
                    start_offset: state.viewport.offset,
                    start_cursor: Vec2::new(cx, cy),
                });
            }

            CursorMoved { .. } => {
                match &state.interaction {
                    Some(ActiveInteraction::Panning { start_offset, start_cursor }) => {
                        let dx = (cx - start_cursor.x) / state.viewport.zoom;
                        let dy = (cy - start_cursor.y) / state.viewport.zoom;
                        state.viewport.offset = Vec2::new(start_offset.x + dx, start_offset.y + dy);
                    }
                    Some(ActiveInteraction::Dragging { node, start_pos }) => {
                        let canvas_pos = state.viewport.to_canvas(Vec2::new(cx, cy));
                        gc.move_node(*node, canvas_pos);
                    }
                    Some(ActiveInteraction::Connecting { from_node, from_pin, .. }) => {
                        state.interaction = Some(ActiveInteraction::Connecting {
                            from_node: *from_node,
                            from_pin: from_pin.clone(),
                            cursor: Vec2::new(cx, cy),
                        });
                    }
                    None => {
                        // 更新 hover
                        state.hover = Self::hit_test(cx, cy, graph, nm, &state.viewport);
                    }
                }
            }

            ButtonReleased(Button::Left) | ButtonReleased(Button::Middle) => {
                if let Some(ActiveInteraction::Connecting { from_node, from_pin, .. }) = &state.interaction {
                    // 检查是否松手在输入引脚上
                    if let Some((to_node, to_pin)) = Self::hit_input_pin(cx, cy, graph, nm, &state.viewport) {
                        let conn = Connection {
                            from_node: *from_node,
                            from_pin: from_pin.clone(),
                            to_node,
                            to_pin,
                        };
                        let _ = gc.connect(conn);
                    }
                }
                state.interaction = None;
            }

            WheelScrolled { delta } => {
                let dy = match delta {
                    iced::mouse::ScrollDelta::Lines { y, .. } => *y,
                    iced::mouse::ScrollDelta::Pixels { y, .. } => *y / 100.0,
                };
                let factor = if dy > 0.0 { 1.1 } else { 1.0 / 1.1 };
                state.viewport.zoom_at(Vec2::new(cx, cy), factor);
            }

            _ => {}
        }
    }

    fn hit_test(
        cx: f32, cy: f32,
        graph: &Graph,
        nm: &Arc<NodeManager>,
        vp: &Viewport,
    ) -> Option<HoverTarget> {
        // 优先检查引脚
        if let Some((id, name)) = Self::hit_input_pin(cx, cy, graph, nm, vp) {
            return Some(HoverTarget::InputPin(id, name));
        }
        if let Some((id, name)) = Self::hit_output_pin(cx, cy, graph, nm, vp) {
            return Some(HoverTarget::OutputPin(id, name));
        }
        if let Some(id) = Self::hit_node(cx, cy, graph, nm, vp) {
            return Some(HoverTarget::Node(id));
        }
        None
    }

    fn hit_node(
        cx: f32, cy: f32,
        graph: &Graph,
        nm: &Arc<NodeManager>,
        vp: &Viewport,
    ) -> Option<NodeId> {
        for (id, node) in &graph.nodes {
            if let Some(def) = nm.get(&node.type_id) {
                let bounds = node_bounds(node.position, def, vp);
                if bounds.contains(iced::Point::new(cx, cy)) {
                    return Some(*id);
                }
            }
        }
        None
    }

    fn hit_input_pin(
        cx: f32, cy: f32,
        graph: &Graph,
        nm: &Arc<NodeManager>,
        vp: &Viewport,
    ) -> Option<(NodeId, String)> {
        let threshold = PIN_RADIUS * vp.zoom * 2.0;
        for (id, node) in &graph.nodes {
            if let Some(def) = nm.get(&node.type_id) {
                for (i, pin) in def.inputs.iter().enumerate() {
                    let pos = input_pin_position(node.position, i, vp);
                    let dx = cx - pos.x;
                    let dy = cy - pos.y;
                    if dx * dx + dy * dy < threshold * threshold {
                        return Some((*id, pin.name.clone()));
                    }
                }
            }
        }
        None
    }

    fn hit_output_pin(
        cx: f32, cy: f32,
        graph: &Graph,
        nm: &Arc<NodeManager>,
        vp: &Viewport,
    ) -> Option<(NodeId, String)> {
        let threshold = PIN_RADIUS * vp.zoom * 2.0;
        for (id, node) in &graph.nodes {
            if let Some(def) = nm.get(&node.type_id) {
                for (i, pin) in def.outputs.iter().enumerate() {
                    let pos = output_pin_position(node.position, i, def.inputs.len(), vp);
                    let dx = cx - pos.x;
                    let dy = cy - pos.y;
                    if dx * dx + dy * dy < threshold * threshold {
                        return Some((*id, pin.name.clone()));
                    }
                }
            }
        }
        None
    }
}
```

```rust
    let nodes_with_output: HashSet<NodeId> = self.connections.iter()
```

```rust
pub async fn evaluate_all(&self) -> Result<HashMap<NodeId, HashMap<String, Value>>, Box<dyn std::error::Error + Send + Sync>> {
    let graph = self.graph.snapshot();
```

```rust
use iced::widget::canvas::Frame;
use iced::{Point, Color};

use crate::canvas::Viewport;
use crate::render::{TEXT_SECONDARY, TEXT_PRIMARY, BORDER};

pub const SLIDER_HEIGHT: f32 = 28.0;
pub const TRACK_HEIGHT: f32 = 4.0;
pub const THUMB_RADIUS: f32 = 7.0;

pub struct SliderLayout {
    pub x: f32,
    pub y: f32,
    pub width: f32,
    pub zoom: f32,
}

impl SliderLayout {
    /// 绘制 Slider
    pub fn draw(&self, frame: &mut Frame, label: &str, value: f32, min: f32, max: f32) {
        let z = self.zoom;
        let pct = ((value - min) / (max - min)).clamp(0.0, 1.0);
        let track_y = self.y + 18.0 * z;
        let track_w = self.width - 20.0 * z;
        let track_x = self.x + 10.0 * z;

        // 标签
        frame.fill_text(iced::widget::canvas::Text {
            content: label.to_string(),
            position: Point::new(self.x, self.y + 2.0 * z),
            size: (11.0 * z).into(),
            color: TEXT_SECONDARY,
            ..Default::default()
        });

        // 值
        frame.fill_text(iced::widget::canvas::Text {
            content: format!("{:.2}", value),
            position: Point::new(self.x + self.width - 10.0 * z, self.y + 2.0 * z),
            size: (10.0 * z).into(),
            color: TEXT_PRIMARY,
            horizontal_alignment: iced::alignment::Horizontal::Right,
            ..Default::default()
        });

        // 轨道背景
        let track_bg = iced::widget::canvas::Path::rounded_rectangle(
            Point::new(track_x, track_y),
            iced::Size::new(track_w, TRACK_HEIGHT * z),
            (2.0 * z).into(),
        );
        frame.fill(&track_bg, BORDER);

        // 填充
        let fill_w = track_w * pct;
        if fill_w > 0.0 {
            let fill = iced::widget::canvas::Path::rounded_rectangle(
                Point::new(track_x, track_y),
                iced::Size::new(fill_w, TRACK_HEIGHT * z),
                (2.0 * z).into(),
            );
            frame.fill(&fill, TEXT_PRIMARY);
        }

        // Thumb
        let thumb_x = track_x + fill_w;
        let thumb_y = track_y + TRACK_HEIGHT * z / 2.0;
        let thumb = iced::widget::canvas::Path::circle(
            Point::new(thumb_x, thumb_y),
            THUMB_RADIUS * z,
        );
        frame.fill(&thumb, Color::WHITE);
        let thumb_border = iced::widget::canvas::Stroke::default()
            .with_color(TEXT_PRIMARY)
            .with_width(1.5);
        frame.stroke(&thumb, thumb_border);
    }

    /// 测试点击是否在 Slider 轨道/Thumb 区域内
    pub fn hit_test(&self, cx: f32, cy: f32) -> bool {
        let z = self.zoom;
        let track_y = self.y + 14.0 * z;
        cx >= self.x && cx <= self.x + self.width
            && cy >= track_y && cy <= track_y + SLIDER_HEIGHT * z
    }

    /// 从屏幕 x 坐标计算 Slider 值
    pub fn value_from_x(&self, cx: f32, min: f32, max: f32) -> f32 {
        let z = self.zoom;
        let track_x = self.x + 10.0 * z;
        let track_w = self.width - 20.0 * z;
        let pct = ((cx - track_x) / track_w).clamp(0.0, 1.0);
        min + pct * (max - min)
    }
}
```
